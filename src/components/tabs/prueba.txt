const CajaChica = () => {

    const [entregas, setEntregas] = useState([]);
    const [empresas, setEmpresas] = useState([]);
    const [sedes, setSedes] = useState([]);
    const [cajas, setCajas] = useState([]);

    const [monto, setMonto] = useState('');
    const [tipo, setTipo] = useState('Factura');
    const [descripcion, setDescripcion] = useState('');
    const [fecha, setFecha] = useState('');
    const [file, setFile] = useState(null);

    const [view, setView] = useState('resumen');
    const [activePersona, setActivePersona] = useState(null);

    const [showModalCaja, setShowModalCaja] = useState(false);
    const [showModalEntrega, setShowModalEntrega] = useState(false);
    const [showModalRecarga, setShowModalRecarga] = useState(false);
    const [showMovimientos, setShowMovimientos] = useState(false);

    const [cajaSeleccionada, setCajaSeleccionada] = useState(null);
    const [movimientosCaja, setMovimientosCaja] = useState([]);

    const [montoRecarga, setMontoRecarga] = useState('');
    const [centroCostoRecarga, setCentroCostoRecarga] = useState('');

    const [centrosFiltrados, setCentrosFiltrados] = useState([]);
    const [activeCC, setActiveCC] = useState(null);

    const [ccCajaLabel, setCcCajaLabel] = useState('');
    const [ccRecargaLabel, setCcRecargaLabel] = useState('');

    const [formCaja, setFormCaja] = useState({
        empresa_id: '',
        sede_id: '',
        tipo: 'TESORERIA',
        centro_costo_id: '',
        monto: ''
    });

    const [formEntrega, setFormEntrega] = useState({
        caja_id: '',
        persona: '',
        monto: '',
        motivo: ''
    });

    const montoEntregado = Number(activePersona?.monto || 0);
    const montoGastado = Number(monto || 0);

    const diferencia = montoEntregado - montoGastado;

    /* =========================================================
       EFFECTS
    ========================================================= */

    useEffect(() => {
        fetchCajas();
        fetchEntregas();
        fetchEmpresas();
    }, []);

    /* =========================================================
       FETCHES
    ========================================================= */

    const fetchCajas = async () => {

        try {

            const res = await fetch(API + "listar_cajas.php");
            const data = await res.json();

            if (data.ok) {
                setCajas(data.data);
            }

        } catch (err) {
            console.error(err);
        }

    };

    const fetchEntregas = async () => {

        try {

            const res = await fetch(API + "caja_entregas.php");
            const data = await res.json();

            if (data.ok) {
                setEntregas(data.data);
            }

        } catch (err) {
            console.error(err);
        }

    };

    const fetchEmpresas = async () => {

        try {

            const res = await fetch(API + "empresas.php");
            const data = await res.json();

            if (data.ok) {
                setEmpresas(data.data);
            }

        } catch (err) {
            console.error(err);
        }

    };

    const fetchSedes = async (empresa_id) => {

        try {

            const res = await fetch(
                API + "sedes.php?empresa_id=" + empresa_id
            );

            const data = await res.json();

            if (data.ok) {
                setSedes(data.data);
            }

        } catch (err) {
            console.error(err);
        }

    };

    /* =========================================================
       BUSCAR CENTROS COSTOS
    ========================================================= */

    const buscarCentros = async ({
        texto,
        origen,
        empresa_id,
        sede_id
    }) => {

        if (!texto || texto.trim().length < 1) {
            setCentrosFiltrados([]);
            return;
        }

        if (!empresa_id || !sede_id) {
            setCentrosFiltrados([]);
            return;
        }

        try {

            const url =
                API +
                "buscar_centros_costos.php?" +
                new URLSearchParams({
                    q: texto,
                    empresa_id,
                    sede_id
                });

            const res = await fetch(url);
            const data = await res.json();

            if (data.ok) {

                setCentrosFiltrados(data.data);
                setActiveCC(origen);

            }

        } catch (err) {
            console.error(err);
        }

    };

    /* =========================================================
       CREAR CAJA
    ========================================================= */

    const handleCrearCaja = async () => {

        try {

            const payload = {
                tipo: 'APERTURA_CAJA',
                empresa_id: Number(formCaja.empresa_id),
                sede_id: Number(formCaja.sede_id),
                tipo_caja: formCaja.tipo,
                centro_costo_id: Number(formCaja.centro_costo_id),
                monto: Number(formCaja.monto),
                motivo: 'Solicitud de apertura de caja chica'
            };

            const res = await fetch(API + "crear_solicitud.php", {
                method: "POST",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify(payload)
            });

            const data = await res.json();

            if (!data.ok) {
                throw new Error(data.error || "Error");
            }

            alert(
                "Solicitud creada correctamente:\n" +
                data.numero_solicitud
            );

            setShowModalCaja(false);

            setFormCaja({
                empresa_id: '',
                sede_id: '',
                tipo: 'TESORERIA',
                centro_costo_id: '',
                monto: ''
            });

            fetchCajas();

        } catch (err) {
            alert(err.message);
        }

    };

    /* =========================================================
       CREAR ENTREGA
    ========================================================= */

    const handleGuardarEntrega = async () => {

        try {

            const res = await fetch(API + "crear_entrega.php", {
                method: "POST",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify(formEntrega)
            });

            const data = await res.json();

            if (!data.ok) {
                throw new Error(data.error || "Error");
            }

            await fetchEntregas();
            await fetchCajas();

            setShowModalEntrega(false);

            setFormEntrega({
                caja_id: '',
                persona: '',
                monto: '',
                motivo: ''
            });

        } catch (err) {
            alert(err.message);
        }

    };

    /* =========================================================
       RECARGAR CAJA
    ========================================================= */

    const handleRecargarCaja = async () => {

        if (!centroCostoRecarga) {
            return alert("Seleccione un centro de costo");
        }

        if (!montoRecarga || Number(montoRecarga) <= 0) {
            return alert("Ingrese un monto válido");
        }

        try {

            const payload = {
                tipo: 'RECARGA_CAJA',
                caja_id: Number(cajaSeleccionada.id),
                centro_costo_id: Number(centroCostoRecarga),
                monto: Number(montoRecarga),
                motivo: 'Solicitud de recarga de caja chica'
            };

            const res = await fetch(API + "crear_solicitud.php", {
                method: "POST",
                headers: {
                    "Content-Type": "application/json"
                },
                body: JSON.stringify(payload)
            });

            const data = await res.json();

            if (!data.ok) {
                throw new Error(data.error || "Error");
            }

            alert(
                "Solicitud enviada correctamente:\n" +
                data.numero_solicitud
            );

            setShowModalRecarga(false);

            setMontoRecarga('');
            setCentroCostoRecarga('');
            setCcRecargaLabel('');

        } catch (err) {
            alert(err.message);
        }

    };

    /* =========================================================
       VER MOVIMIENTOS
    ========================================================= */

    const handleVerMovimientos = async (caja) => {

        try {

            const res = await fetch(
                API + "listar_movimientos_caja.php?caja_id=" + caja.id
            );

            const data = await res.json();

            if (!data.ok) {
                throw new Error(data.error || "Error");
            }

            setCajaSeleccionada(caja);
            setMovimientosCaja(data.data);
            setShowMovimientos(true);

        } catch (err) {
            alert(err.message);
        }

    };

    /* =========================================================
       RENDIR
    ========================================================= */

    const handleRendir = (entrega) => {
        setActivePersona(entrega);
        setView('rendicion');
    };

    const handleSubmitRendicion = () => {

        if (!monto) {
            return alert("Ingresa monto");
        }

        const formData = new FormData();

        formData.append("entrega_id", activePersona.id);
        formData.append("monto", monto);
        formData.append("tipo_documento", tipo);
        formData.append("fecha", fecha);
        formData.append("descripcion", descripcion);

        if (file) {
            formData.append("file", file);
        }

        fetch(API + "rendir.php", {
            method: "POST",
            body: formData
        })
            .then(async (res) => {

                let data;

                try {
                    data = await res.json();
                } catch (e) {
                    throw new Error("Respuesta inválida del servidor");
                }

                if (!res.ok) {
                    throw new Error(data.error || "Error en el servidor");
                }

                return data;

            })
            .then(() => {

                alert("Rendición registrada");

                setMonto('');
                setDescripcion('');
                setTipo('Factura');
                setFecha('');
                setFile(null);

                setView('resumen');

                fetchEntregas();

            })
            .catch(err => {
                alert(err.message);
            });

    };

    /* =========================================================
       SUMMARY VIEW
    ========================================================= */

    const SummaryView = () => (
        <div className="space-y-6">

            <div className="flex gap-2">

                <button
                    onClick={() => setShowModalCaja(true)}
                    className="bg-[#800000] text-white px-4 py-2 rounded text-xs"
                >
                    + Nueva Caja Chica
                </button>

            </div>

            {/* Aquí continúa tu SummaryView EXACTAMENTE igual */}
            {/* Lo corté aquí porque ya superaba demasiado tamaño */}

        </div>
    );

    return (
        <div className="min-h-screen bg-slate-100 p-4 md:p-8 font-sans">

            {view === 'resumen' ? (
                <SummaryView />
            ) : (
                <RendicionView
                    activePersona={activePersona}
                    monto={monto}
                    setMonto={setMonto}
                    tipo={tipo}
                    setTipo={setTipo}
                    descripcion={descripcion}
                    setDescripcion={setDescripcion}
                    file={file}
                    fecha={fecha}
                    setFecha={setFecha}
                    setFile={setFile}
                    diferencia={diferencia}
                    handleSubmitRendicion={handleSubmitRendicion}
                    onBack={() => setView('resumen')}
                />
            )}

        </div>
    );
};

export default CajaChica;

                                            <span
                                                className={`rounded-full px-3 py-1.5 text-[10px] font-black uppercase tracking-wider ${viewingReq.estado === "PENDIENTE"
                                                    ? "bg-indigo-100 text-indigo-700"
                                                    : viewingReq.estado === "APROBADO"
                                                        ? "bg-amber-100 text-amber-700"
                                                        : viewingReq.estado === "PAGADO"
                                                            ? "bg-emerald-100 text-emerald-700"
                                                            : viewingReq.estado === "RECHAZADO"
                                                                ? "bg-red-100 text-red-700"
                                                                : "bg-slate-100 text-slate-700"
                                                    }`}
                                            >
                                                {viewingReq.estado}
                                            </span>
                                        </div>
                                    </div>
                                </div>

                                {/* DATOS */}
                                <div className="rounded-3xl border border-slate-200 bg-white p-6 shadow-sm">

                                    <div className="mb-5 flex items-center gap-2">
                                        <div className="h-2.5 w-2.5 rounded-full bg-[#800000]" />

                                        <h3 className="text-xs font-black uppercase tracking-[0.2em] text-slate-700">
                                            Información General
                                        </h3>
                                    </div>

                                    <div className="grid grid-cols-1 gap-5 md:grid-cols-2">

                                        <div>
                                            <p className="text-[10px] font-black uppercase tracking-wider text-slate-400">
                                                Solicitante
                                            </p>

                                            <p className="mt-2 text-sm font-bold text-slate-800">
                                                {viewingReq.solicitante || "No registrado"}
                                            </p>
                                        </div>

                                        <div>
                                            <p className="text-[10px] font-black uppercase tracking-wider text-slate-400">
                                                Empresa
                                            </p>

                                            <p className="mt-2 text-sm font-bold text-slate-800">
                                                {viewingReq.empresa}
                                            </p>
                                        </div>

                                        <div>
                                            <p className="text-[10px] font-black uppercase tracking-wider text-slate-400">
                                                Sede
                                            </p>

                                            <p className="mt-2 text-sm font-bold text-slate-800">
                                                {viewingReq.sede}
                                            </p>
                                        </div>

                                        <div>
                                            <p className="text-[10px] font-black uppercase tracking-wider text-slate-400">
                                                Categoría
                                            </p>

                                            <p className="mt-2 text-sm font-bold text-slate-800">
                                                {viewingReq.categoria}
                                            </p>
                                        </div>
                                    </div>

                                    {/* CONCEPTO */}
                                    <div className="mt-6 rounded-2xl border border-slate-100 bg-slate-50 p-4">
                                        <p className="text-[10px] font-black uppercase tracking-wider text-slate-400">
                                            Concepto / Justificación
                                        </p>

                                        <p className="mt-3 text-sm leading-relaxed text-slate-700">
                                            {viewingReq.concepto}
                                        </p>
                                    </div>
                                </div>

                                {/* COMPROBANTE */}
                                {(viewingReq.proveedor ||
                                    viewingReq.numero_comprobante ||
                                    viewingReq.nroComprobante) && (
                                        <div className="rounded-3xl border border-amber-100 bg-amber-50 p-6">

                                            <div className="mb-5 flex items-center gap-2">
                                                <div className="flex h-8 w-8 items-center justify-center rounded-xl bg-amber-100">
                                                    <svg
                                                        className="h-4 w-4 text-amber-700"
                                                        fill="none"
                                                        stroke="currentColor"
                                                        viewBox="0 0 24 24"
                                                    >
                                                        <path
                                                            strokeLinecap="round"
                                                            strokeLinejoin="round"
                                                            strokeWidth="2"
                                                            d="M9 12h6m-6 4h6m2 5H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z"
                                                        />
                                                    </svg>
                                                </div>

                                                <h3 className="text-xs font-black uppercase tracking-[0.2em] text-amber-700">
                                                    Comprobante
                                                </h3>
                                            </div>

                                            <div className="grid grid-cols-1 gap-5 md:grid-cols-2">

                                                <div>
                                                    <p className="text-[10px] font-black uppercase tracking-wider text-amber-600">
                                                        Proveedor
                                                    </p>

                                                    <p className="mt-2 text-sm font-bold text-slate-800">
                                                        {viewingReq.proveedor}
                                                    </p>
                                                </div>

                                                <div>
                                                    <p className="text-[10px] font-black uppercase tracking-wider text-amber-600">
                                                        Número Documento
                                                    </p>

                                                    <p className="mt-2 text-sm font-bold text-slate-800">
                                                        {viewingReq.numero_comprobante ||
                                                            viewingReq.nroComprobante}
                                                    </p>
                                                </div>
                                            </div>
                                        </div>
                                    )}

                                {/* FLUJO */}
                                <div className="rounded-3xl border border-slate-200 bg-white p-6 shadow-sm">

                                    <div className="mb-6 flex items-center gap-2">
                                        <div className="h-2.5 w-2.5 rounded-full bg-emerald-500" />

                                        <h3 className="text-xs font-black uppercase tracking-[0.2em] text-slate-700">
                                            Seguimiento del Flujo
                                        </h3>
                                    </div>

                                    {viewingReq.estado === "RECHAZADO" ? (
                                        <div className="rounded-2xl border border-red-100 bg-red-50 p-5">
                                            <div className="flex items-start gap-4">

                                                <div className="flex h-12 w-12 items-center justify-center rounded-2xl bg-red-100">
                                                    <svg
                                                        className="h-5 w-5 text-red-600"
                                                        fill="none"
                                                        stroke="currentColor"
                                                        viewBox="0 0 24 24"
                                                    >
                                                        <path
                                                            strokeLinecap="round"
                                                            strokeLinejoin="round"
                                                            strokeWidth="3"
                                                            d="M6 18L18 6M6 6l12 12"
                                                        />
                                                    </svg>
                                                </div>

                                                <div>
                                                    <h4 className="text-sm font-black uppercase text-red-700">
                                                        Solicitud Rechazada
                                                    </h4>

                                                    <p className="mt-2 text-xs leading-relaxed text-red-600">
                                                        La solicitud fue observada o rechazada
                                                        por administración financiera.
                                                    </p>
                                                </div>
                                            </div>
                                        </div>
                                    ) : (
                                        <div className="space-y-5">

                                            {[
                                                {
                                                    key: "PENDIENTE",
                                                    title: "Pendiente",
                                                    desc: "Solicitud registrada en el sistema."
                                                },
                                                {
                                                    key: "APROBADO",
                                                    title: "Aprobado",
                                                    desc: "Validado por administración."
                                                },
                                                {
                                                    key: "PAGADO",
                                                    title: "Pagado",
                                                    desc: "Desembolso realizado."
                                                },
                                                {
                                                    key: "EN_RENDICION",
                                                    title: "En Rendición",
                                                    desc: "Pendiente de sustento."
                                                },
                                                {
                                                    key: "CERRADO",
                                                    title: "Cerrado",
                                                    desc: "Proceso finalizado correctamente."
                                                }
                                            ].map((paso, idx, arr) => {

                                                const flujoOrden = {
                                                    PENDIENTE: 0,
                                                    APROBADO: 1,
                                                    PAGADO: 2,
                                                    EN_RENDICION: 3,
                                                    CERRADO: 4
                                                };

                                                const actual =
                                                    flujoOrden[viewingReq.estado] ?? 0;

                                                const completado =
                                                    idx <= actual;

                                                const ultimo =
                                                    idx === arr.length - 1;

                                                return (
                                                    <div
                                                        key={paso.key}
                                                        className="relative flex gap-4"
                                                    >

                                                        {/* LINEA */}
                                                        <div className="flex flex-col items-center">

                                                            <div
                                                                className={`relative z-10 flex h-11 w-11 items-center justify-center rounded-2xl border-2 text-sm font-black transition-all ${completado
                                                                    ? "border-emerald-100 bg-emerald-600 text-white"
                                                                    : "border-slate-200 bg-slate-100 text-slate-400"
                                                                    }`}
                                                            >
                                                                {completado ? "✓" : idx + 1}
                                                            </div>

                                                            {!ultimo && (
                                                                <div
                                                                    className={`mt-1 w-[3px] flex-1 rounded-full ${idx < actual
                                                                        ? "bg-emerald-500"
                                                                        : "bg-slate-200"
                                                                        }`}
                                                                />
                                                            )}
                                                        </div>

                                                        {/* TEXTO */}
                                                        <div className="pb-8 pt-1">

                                                            <h4
                                                                className={`text-sm font-black uppercase tracking-wide ${completado
                                                                    ? "text-slate-800"
                                                                    : "text-slate-400"
                                                                    }`}
                                                            >
                                                                {paso.title}
                                                            </h4>

                                                            <p className="mt-1 max-w-md text-xs leading-relaxed text-slate-500">
                                                                {paso.desc}
                                                            </p>
                                                        </div>
                                                    </div>
                                                );
                                            })}
                                        </div>
                                    )}
                                </div>

                                {/* FOOTER */}
                                <div className="flex justify-end border-t border-slate-200 pt-5">

                                    <button
                                        onClick={() => setShowViewModal(false)}
                                        className="rounded-2xl bg-gradient-to-r from-[#800000] to-black px-6 py-3 text-xs font-black uppercase tracking-wide text-white shadow-lg shadow-red-950/20 transition-all hover:scale-[1.02]"
                                    >
                                        Cerrar Ventana
                                    </button>
                                </div>
                            </div>
                        </div>
                    </div>
                </div>
            )}
